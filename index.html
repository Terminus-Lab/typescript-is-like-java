Basics

Hello world · Variables · Types · Control flow

Hello World

TypeScript
console.log("Hello, world!");
Java
public class Main {
    public static void main(String[] args) {
        System.out.println("Hello, world!");

Variables and Constants

TypeScript
let myVariable = 42;
myVariable = 50;

const myConstant = 42;
Java
var myVariable = 42;
myVariable = 50;

final var myConstant = 42;

Explicit Types

TypeScript
const explicitDouble: number = 70.0;
const explicitString: string = "Hello";
Java
double explicitDouble = 70.0;
String explicitString = "Hello";

Type Coercion

TypeScript
const label = "The width is ";
const width = 94;
const widthLabel = label + width;
// "The width is 94"
Java
var label = "The width is ";
var width = 94;
var widthLabel = label + width;
// "The width is 94"

String Interpolation

Java has no built-in string interpolation. Use String.format or concatenation.

TypeScript
const apples = 3;
const oranges = 5;
const fruitSummary =
    `I have ${apples + oranges} pieces of fruit.`;
Java
var apples = 3;
var oranges = 5;
var fruitSummary = String.format(
    "I have %d pieces of fruit.", apples + oranges);

// Or simple concatenation
var summary2 = "I have " + (apples + oranges) + " pieces.";

If Statement

TypeScript
const age = 42;

if (age < 10) {
    console.log("You're too young to watch this movie");
} else if (age < 13) {
    console.log("You can watch this movie with a parent");
} else {
    console.log("You can watch this movie");
}
Java
var age = 42;

if (age < 10) {
    System.out.println("You're too young to watch this movie");
} else if (age < 13) {
    System.out.println("You can watch this movie with a parent");
} else {
    System.out.println("You can watch this movie");
}

Ternary

TypeScript
const loaded = true;
const status = loaded ? "Ready" : "Loading...";
// "Ready"
Java
var loaded = true;
var status = loaded ? "Ready" : "Loading...";
// "Ready"

Switch / Pattern Matching

TypeScript
const nb = 42;
let text: string;

switch (true) {
    case nb < 10:    text = "single digit"; break;
    case nb < 100:   text = "double digits"; break;
    case nb < 1000:  text = "triple digits"; break;
    default:         text = "four or more digits";
}
// "double digits"
Java
var nb = 42;
// Switch expression with pattern matching (Java 21+)
var text = switch (nb) {
    case Integer n when n < 10   -> "single digit";
    case Integer n when n < 100  -> "double digits";
    case Integer n when n < 1000 -> "triple digits";
    default                      -> "four or more digits";
};
// "double digits"
02

Functions

Definitions · Lambdas · Higher-order · Tuples

Functions

TypeScript
function greet(name: string, day: string): string {
    return `Hello ${name}, today is ${day}.`;
}

const text = greet("Bob", "Tuesday");
// Hello Bob, today is Tuesday.
Java
String greet(String name, String day) {
    return String.format("Hello %s, today is %s.", name, day);
}

var text = greet("Bob", "Tuesday");
// Hello Bob, today is Tuesday.

Single-Expression Functions

TypeScript
const double = (value: number) => value * 2;

const result = double(4);
// 8
Java
// Java requires a Function reference
Function<Integer, Integer> doubleIt = value -> value * 2;

var result = doubleIt.apply(4);
// 8

Default Arguments

Java has no default arguments. Use overloading to simulate.

TypeScript
function displayGreeting(message: string, name: string = "Guest") {
    console.log(`Hello ${name}, ${message}`);
}

displayGreeting("welcome!");
// Hello Guest, welcome!
Java
void displayGreeting(String message, String name) {
    System.out.println("Hello " + name + ", " + message);
}

void displayGreeting(String message) {
    displayGreeting(message, "Guest");
}

displayGreeting("welcome!");
// Hello Guest, welcome!

Optional Parameters

TypeScript
function greet(name: string, greeting?: string) {
    console.log(`${greeting ?? "Hello"} ${name}`);
}

greet("Anna");          // Hello Anna
greet("Anna", "Hi");    // Hi Anna
Java
// Use overloading or Optional<T>
void greet(String name) {
    greet(name, "Hello");
}

void greet(String name, String greeting) {
    System.out.println(greeting + " " + name);
}

greet("Anna");        // Hello Anna
greet("Anna", "Hi");  // Hi Anna

Variable Number of Arguments

TypeScript
function sumOf(...numbers: number[]): number {
    return numbers.reduce((a, b) => a + b, 0);
}

const sum = sumOf(42, 597, 12);
// 651
Java
int sumOf(int... numbers) {
    return Arrays.stream(numbers).sum();
}

var sum = sumOf(42, 597, 12);
// 651

Higher-Order Function — Return a Function

TypeScript
function makeIncrementer(): (n: number) => number {
    return n => n + 1;
}

const increment = makeIncrementer();
const result = increment(7);
// 8
Java
Function<Integer, Integer> makeIncrementer() {
    return n -> n + 1;
}

var increment = makeIncrementer();
var result = increment.apply(7);
// 8

Higher-Order Function — Function as Parameter

TypeScript
function transform(initial: string, f: (s: string) => string) {
    return f(initial);
}

const result = transform("hello", x => x.toUpperCase());
// HELLO
Java
String transform(String initial, Function<String, String> f) {
    return f.apply(initial);
}

var result = transform("hello", String::toUpperCase);
// HELLO

Tuple / Multi-Value Return

TypeScript has tuples and destructuring. Java uses records.

TypeScript
function getGasPrices(): [number, number, number] {
    return [3.59, 3.69, 3.79];
}

const [a, b, c] = getGasPrices();
Java
record GasPrices(double a, double b, double c) {}

GasPrices getGasPrices() {
    return new GasPrices(3.59, 3.69, 3.79);
}

// Record pattern destructuring (Java 21+)
var prices = getGasPrices();
if (prices instanceof GasPrices(var a, var b, var c)) {
    // use a, b, c
}

Collections

Arrays · Lists · Maps · Sets · Iteration

Arrays

TypeScript
const shoppingList = ["catfish", "water", "tulips", "blue paint"];
shoppingList[1] = "bottle of water";
Java
String[] shoppingList = {"catfish", "water", "tulips", "blue paint"};
shoppingList[1] = "bottle of water";

Lists

TypeScript arrays are dynamic. Java distinguishes immutable from mutable lists.

TypeScript
// Immutable (by convention)
const readonly: readonly string[] =
    ["catfish", "water", "tulips"];

// Mutable
const list: string[] = ["catfish", "water", "tulips"];
list[1] = "bottle of water";
list.push("bucket");
Java
// Immutable
List<String> readonly = List.of("catfish", "water", "tulips");

// Mutable
var list = new ArrayList<>(List.of("catfish", "water", "tulips"));
list.set(1, "bottle of water");
list.add("bucket");

Maps / Dictionaries

TypeScript
const occupations: Record<string, string> = {
    "Malcolm": "Captain",
    "Kaylee": "Mechanic"
};

// Or Map class
const map = new Map<string, string>([
    ["Malcolm", "Captain"],
    ["Kaylee", "Mechanic"]
]);
map.set("Jayne", "Public Relations");
Java
// Immutable
Map<String, String> occupations = Map.of(
    "Malcolm", "Captain",
    "Kaylee", "Mechanic"
);

// Mutable
var map = new HashMap<String, String>();
map.put("Malcolm", "Captain");
map.put("Kaylee", "Mechanic");
map.put("Jayne", "Public Relations");

Sets

TypeScript
const tags = new Set<string>(["a", "b", "c"]);
tags.add("d");
tags.has("a"); // true
Java
var tags = new HashSet<>(Set.of("a", "b", "c"));
tags.add("d");
tags.contains("a"); // true

Empty Collections

TypeScript
const emptyArray: string[] = [];
const emptyMap: Record<string, number> = {};
const emptySet = new Set<string>();
Java
List<String> emptyList = new ArrayList<>();
Map<String, Integer> emptyMap = new HashMap<>();
Set<String> emptySet = new HashSet<>();

// Read-only
List<String> immutable = List.of();

For-Each Iteration

TypeScript

for (const name of names) {
    console.log(`Person is called ${name}`);
}

names.forEach(name =>
    console.log(`Person is called ${name}`));
Java
var names = List.of("Anna", "Alex", "Brian", "Jack");

for (var name : names) {
    System.out.println("Person is called " + name);
}

names.forEach(name ->
    System.out.println("Person is called " + name));

Indexed Iteration

TypeScript
const names = ["Anna", "Alex", "Brian", "Jack"];

for (let i = 0; i < names.length; i++) {
    console.log(`Person ${i + 1} is called ${names[i]}`);
}

// Or with entries()
for (const [i, name] of names.entries()) {
    console.log(`Person ${i + 1} is called ${name}`);
}
Java
var names = List.of("Anna", "Alex", "Brian", "Jack");

for (int i = 0; i < names.size(); i++) {
    System.out.printf("Person %d is called %s%n", i + 1, names.get(i));
}

// Or with IntStream
IntStream.range(0, names.size()).forEach(i ->
    System.out.printf("Person %d is called %s%n", i + 1, names.get(i)));

Slicing

TypeScript
const names = ["Anna", "Alex", "Brian", "Jill", "Jack"];

// From index 1 up to (but not including) the last
const middle = names.slice(1, -1);
// ["Alex", "Brian", "Jill"]
Java
var names = List.of("Anna", "Alex", "Brian", "Jill", "Jack");

// subList(fromInclusive, toExclusive)
var middle = names.subList(1, names.size() - 1);
// [Alex, Brian, Jill]

Operations

Map · Filter · Reduce · Sort · Streams

Map

TypeScript
const numbers = [20, 19, 7, 12];
const multiplied = numbers.map(n => 3 * n);
// [60, 57, 21, 36]
Java
var numbers = List.of(20, 19, 7, 12);
var multiplied = numbers.stream()
    .map(n -> 3 * n)
    .toList();
// [60, 57, 21, 36]

Filter

TypeScript
const numbers = [1, 2, 3, 4, 5];
const evens = numbers.filter(n => n % 2 === 0);
// [2, 4]
Java
var numbers = List.of(1, 2, 3, 4, 5);
var evens = numbers.stream()
    .filter(n -> n % 2 == 0)
    .toList();
// [2, 4]

Sort

TypeScript
// Note: default sort is lexicographic, even for numbers
const ordered = [1, 5, 3, 12, 2].sort((a, b) => a - b);
// [1, 2, 3, 5, 12]
Java
var ordered = Stream.of(1, 5, 3, 12, 2)
    .sorted()
    .toList();
// [1, 2, 3, 5, 12]

Reduce

TypeScript
const total = [1, 2, 3, 4].reduce((acc, n) => acc + n, 0);
// 10
Java
var total = Stream.of(1, 2, 3, 4)
    .reduce(0, Integer::sum);
// 10

Filter / Group By / Average

TypeScript
type SensorData = { id: number; location: string; value: number };

const data: SensorData[] = [
    { id: 1, location: "A", value: 2.89 },
    { id: 2, location: "B", value: 12.01 },
    { id: 3, location: "B", value: 11.89 },
    { id: 4, location: "A", value: 3.11 },
    { id: 5, location: "A", value: -456.0 }
];

const grouped = data
    .filter(d => d.value > -50)
    .reduce<Record<string, number[]>>((acc, d) => {
        (acc[d.location] ??= []).push(d.value);
        return acc;
    }, {});

const avgs = Object.entries(grouped).map(([location, values]) => ({
    location,
    avg: values.reduce((a, b) => a + b, 0) / values.length
}));

// [{ location: "A", avg: 3.0 }, { location: "B", avg: 11.95 }]
Java
record SensorData(int id, String location, double value) {}
record LocationAvg(String location, double avg) {}

var data = List.of(
    new SensorData(1, "A", 2.89),
    new SensorData(2, "B", 12.01),
    new SensorData(3, "B", 11.89),
    new SensorData(4, "A", 3.11),
    new SensorData(5, "A", -456.0)
);

var avgs = data.stream()
    .filter(d -> d.value() > -50.0)
    .collect(Collectors.groupingBy(
        SensorData::location,
        Collectors.averagingDouble(SensorData::value)))
    .entrySet().stream()
    .map(e -> new LocationAvg(e.getKey(), e.getValue()))
    .toList();

// [LocationAvg[location=A, avg=3.0],
//  LocationAvg[location=B, avg=11.95]]

Lazy Evaluation

TypeScript array methods evaluate eagerly. Java streams are lazy until a terminal operation.

TypeScript
// Eager: every filter and map runs immediately
const items = [1, 2, 3, 4]
    .filter(n => { console.log(`Filter ${n}`); return n % 2 === 1; })
    .map(n => { console.log(`Map ${n}`); return n * 2; });

// Filter 1, Filter 2, Filter 3, Filter 4
// Map 1, Map 3
Java
// Lazy: nothing runs until toList()
var items = Stream.of(1, 2, 3, 4)
    .filter(n -> { System.out.println("Filter " + n); return n % 2 == 1; })
    .map(n -> { System.out.println("Map " + n); return n * 2; })
    .toList();

// Filter 1, Map 1,
// Filter 2,
// Filter 3, Map 3,
// Filter 4

Declaration

TypeScript
class Shape {
    numberOfSides = 0;

        return `A shape with ${this.numberOfSides} sides.`;
    }
}
Java
class Shape {
    int numberOfSides = 0;

    String simpleDescription() {
        return "A shape with " + numberOfSides + " sides.";
    }
}

Usage

TypeScript
const shape = new Shape();
shape.numberOfSides = 7;
const desc = shape.simpleDescription();
// A shape with 7 sides.
Java
var shape = new Shape();
shape.numberOfSides = 7;
var desc = shape.simpleDescription();
// A shape with 7 sides.

Subclass

TypeScript
class NamedShape {
    numberOfSides = 0;
    constructor(public name: string) {}

    simpleDescription(): string {
        return `A shape with ${this.numberOfSides} sides.`;
    }
}

class Square extends NamedShape {
    constructor(public sideLength: number, name: string) {
        super(name);
        this.numberOfSides = 4;
    }

    area(): number {
        return this.sideLength ** 2;
    }

    simpleDescription(): string {
        return `A square with sides of length ${this.sideLength}.`;
    }
}

const square = new Square(5.2, "My square");
Java
class NamedShape {
    protected int numberOfSides = 0;
    protected final String name;

    NamedShape(String name) { this.name = name; }

    String simpleDescription() {
        return "A shape with " + numberOfSides + " sides.";
    }
}

class Square extends NamedShape {
    private final double sideLength;

    Square(double sideLength, String name) {
        super(name);
        this.sideLength = sideLength;
        this.numberOfSides = 4;
    }

    double area() { return Math.pow(sideLength, 2); }

    @Override
    String simpleDescription() {
        return "A square with sides of length " + sideLength + ".";
    }
}

var square = new Square(5.2, "My square");

TypeScript

Java

TypeScript

Casting

TypeScript
// Unsafe (compile-time only — no runtime check)
const circle = shape as Circle;

// Safe with type guard

    const circle: Circle = shape; // narrowed
}
Java

Circle circle = (Circle) shape;

// Safe with pattern matching

TypeScript

Union Types / Sealed Interfaces

TypeScript has structural unions. Java models alternatives with sealed interfaces.

TypeScript
type Shape =
    | { kind: "circle"; radius: number }
    | { kind: "square"; side: number };

function area(s: Shape): number {
    switch (s.kind) {
        case "circle": return Math.PI * s.radius ** 2;
        case "square": return s.side ** 2;
    }
}
Java
sealed interface Shape permits Circle, Square {}
record Circle(double radius) implements Shape {}
record Square(double side) implements Shape {}

double area(Shape s) {
    return switch (s) {
        case Circle c -> Math.PI * Math.pow(c.radius(), 2);
        case Square sq -> Math.pow(sq.side(), 2);
    };
}

Exceptions

Try · Catch · Finally · Checked exceptions

Exception Handling

TypeScript
try {
    // some code
} catch (e) {
    if (e instanceof CustomError && e.code === 404) {
        // handle 404
    } else if (e instanceof Error) {
        // handle other errors
    }
} finally {
    // optional finally block
}
Java
try {
    // some code
} catch (CustomException e) {
    if (e.getCode() == 404) {
        // handle 404
    } else {
        // handle other CustomException
    }
} catch (Exception e) {
    // handle the rest
} finally {
    // optional finally block
}

Throwing Exceptions

Java distinguishes checked exceptions (must be declared or caught) from unchecked. TypeScript doesn't.

TypeScript
function parseAge(input: string): number {
    const n = parseInt(input);
    if (isNaN(n)) {
        throw new Error(`Invalid age: ${input}`);
    }
    return n;
}
Java
int parseAge(String input) throws NumberFormatException {
    try {
        return Integer.parseInt(input);
    } catch (NumberFormatException e) {
        throw new IllegalArgumentException("Invalid age: " + input, e);
    }
}
08

Null

Nullable types · Optional chaining · Coalescing

Nullable Types

TypeScript expresses nullability in the type. Java references are nullable by default — wrap in Optional<T> to make absence explicit.

TypeScript
type Measurement = { celsius: number };

let data: Measurement = null;        // Error in strict mode
let maybe: Measurement | null = null; // OK

function printMaybeNull(data: Measurement | null) {
    if (data === null) return;
    console.log(data.celsius);
}

function printNoNull(data: Measurement) {
    console.log(data.celsius);
}
Java
record Measurement(double celsius) {}

// All references are nullable by default
Measurement data = null; // Always allowed

void printMaybeNull(Measurement data) {
    if (data == null) return;
    System.out.println(data.celsius());
}

// Use Optional<T> to express absence in the type
void printOptional(Optional<Measurement> data) {
    data.ifPresent(d -> System.out.println(d.celsius()));
}

Optional Chaining

TypeScript
type DataPoint = {
    id: number;
    celsius: number;
    child?: DataPoint;
};

const data: DataPoint = {
    id: 1, celsius: 22.1,
    child: { id: 2, celsius: 22.8 }
};

const result = data.child?.child?.celsius;
// undefined
Java
record DataPoint(int id, double celsius, DataPoint child) {}

var data = new DataPoint(1, 22.1,
    new DataPoint(2, 22.8, null));

// No null-safe operator — chain via Optional
var result = Optional.ofNullable(data.child())
    .map(DataPoint::child)
    .map(DataPoint::celsius);
// Optional.empty

Nullish Coalescing

TypeScript
const value: string | null = null;
const result = value ?? "default";
// "default"
Java
String value = null;
var result = value != null ? value : "default";

// Or with Optional
var result2 = Optional.ofNullable(value).orElse("default");
// "default"

Async

Promises · CompletableFuture · Virtual threads

Async / Await

TypeScript has native async/await. Java offers CompletableFuture chains, or — since Java 21 — virtual threads, which let you write linear blocking code that scales like async.

TypeScript
type Stats = { repo: string; stars: number; forks: number };

async function fetchRepo(repo: string): Promise<Stats> {
    const response = await fetch(`https://api.github.com/repos/${repo}`);
    const data = await response.json();
    return { repo, stars: data.stargazers_count, forks: data.forks };
}

const repos = ["microsoft/typescript", "openjdk/jdk"];
const results = await Promise.all(repos.map(fetchRepo));

results.forEach(r =>
    console.log(`${r.repo}: ${r.stars} - ${r.forks}`));
Java
record Stats(String repo, int stars, int forks) {}

CompletableFuture<Stats> fetchRepo(String repo) {
    var client = HttpClient.newHttpClient();
    var request = HttpRequest.newBuilder()
        .uri(URI.create("https://api.github.com/repos/" + repo))
        .build();

    return client.sendAsync(request, BodyHandlers.ofString())
        .thenApply(response -> parseStats(repo, response.body()));
}

var repos = List.of("microsoft/typescript", "openjdk/jdk");
var futures = repos.stream().map(this::fetchRepo).toList();

// Wait for all
var results = futures.stream()
    .map(CompletableFuture::join)
    .toList();

results.forEach(r ->
    System.out.printf("%s: %d - %d%n", r.repo(), r.stars(), r.forks()));
